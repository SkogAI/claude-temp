    36→// wrapCloneError wraps clone errors with helpful suggestions.
    37→// Detects common auth failures and suggests SSH as an alternative.
    38→func wrapCloneError(err error, gitURL string) error {
    39→	errStr := err.Error()
    40→
    41→	// Check for GitHub password auth failure
    42→	if strings.Contains(errStr, "Password authentication is not supported") ||
    43→		strings.Contains(errStr, "Authentication failed") {
    44→		// Check if they used HTTPS
    45→		if strings.HasPrefix(gitURL, "https://") {
    46→			// Try to suggest the SSH equivalent
    47→			sshURL := convertToSSH(gitURL)
    48→			if sshURL != "" {
    49→				return fmt.Errorf("creating bare repo: %w\n\nHint: GitHub no longer supports password authentication.\nTry using SSH instead:\n  gt rig add <name> %s", err, sshURL)
    50→			}
    51→			return fmt.Errorf("creating bare repo: %w\n\nHint: GitHub no longer supports password authentication.\nTry using an SSH URL (git@github.com:owner/repo.git) or a personal access token.", err)
    52→		}
    53→	}
    54→
    55→	return fmt.Errorf("creating bare repo: %w", err)
    56→}

    58→// convertToSSH converts an HTTPS GitHub/GitLab URL to SSH format.
    59→// Returns empty string if conversion is not possible.
    60→func convertToSSH(httpsURL string) string {
    61→	// Handle GitHub: https://github.com/owner/repo.git -> git@github.com:owner/repo.git
    62→	if strings.HasPrefix(httpsURL, "https://github.com/") {
    63→		path := strings.TrimPrefix(httpsURL, "https://github.com/")
    64→		if !strings.HasSuffix(path, ".git") {
    65→			path += ".git"
    66→		}
    67→		return "git@github.com:" + path
    68→	}
    69→
    70→	// Handle GitLab: https://gitlab.com/owner/repo.git -> git@gitlab.com:owner/repo.git
    71→	if strings.HasPrefix(httpsURL, "https://gitlab.com/") {
    72→		path := strings.TrimPrefix(httpsURL, "https://gitlab.com/")
    73→		if !strings.HasSuffix(path, ".git") {
    74→			path += ".git"
    75→		}
    76→		return "git@gitlab.com:" + path
    77→	}
    78→
    79→	return ""
    80→}

   119→// DiscoverRigs returns all rigs registered in the workspace.
   120→// Rigs that fail to load are logged to stderr and skipped; partial results are returned.
   121→func (m *Manager) DiscoverRigs() ([]*Rig, error) {
   122→	var rigs []*Rig
   123→
   124→	for name, entry := range m.config.Rigs {
   125→		rig, err := m.loadRig(name, entry)
   126→		if err != nil {
   127→			fmt.Fprintf(os.Stderr, "Warning: failed to load rig %q: %v\n", name, err)
   128→			continue
   129→		}
   130→		rigs = append(rigs, rig)
   131→	}
   132→
   133→	return rigs, nil
   134→}

   136→// GetRig returns a specific rig by name.
   137→func (m *Manager) GetRig(name string) (*Rig, error) {
   138→	entry, ok := m.config.Rigs[name]
   139→	if !ok {
   140→		return nil, ErrRigNotFound
   141→	}
   142→
   143→	return m.loadRig(name, entry)
   144→}

   148→	_, ok := m.config.Rigs[name]

   152→// loadRig loads rig details from the filesystem.
   153→func (m *Manager) loadRig(name string, entry config.RigEntry) (*Rig, error) {
   154→	rigPath := filepath.Join(m.townRoot, name)
   155→
   156→	// Verify directory exists
   157→	info, err := os.Stat(rigPath)
   158→	if err != nil {
   159→		return nil, fmt.Errorf("rig directory: %w", err)
   160→	}
   161→	if !info.IsDir() {
   162→		return nil, fmt.Errorf("not a directory: %s", rigPath)
   163→	}
   164→
   165→	rig := &Rig{
   166→		Name:      name,
   167→		Path:      rigPath,
   168→		GitURL:    entry.GitURL,
   169→		PushURL:   strings.TrimSpace(entry.PushURL),
   170→		LocalRepo: entry.LocalRepo,
   171→		Config:    entry.BeadsConfig,
   172→	}
   173→
   174→	// Scan for polecats
   175→	polecatsDir := filepath.Join(rigPath, "polecats")
   176→	if entries, err := os.ReadDir(polecatsDir); err == nil {
   177→		for _, e := range entries {
   178→			if !e.IsDir() {
   179→				continue
   180→			}
   181→			name := e.Name()
   182→			if strings.HasPrefix(name, ".") {
   183→				continue
   184→			}
   185→			rig.Polecats = append(rig.Polecats, name)
   186→		}
   187→	}
   188→
   189→	// Scan for crew workers
   190→	crewDir := filepath.Join(rigPath, "crew")
   191→	if entries, err := os.ReadDir(crewDir); err == nil {
   192→		for _, e := range entries {
   193→			if e.IsDir() && !strings.HasPrefix(e.Name(), ".") {
   194→				rig.Crew = append(rig.Crew, e.Name())
   195→			}
   196→		}
   197→	}
   198→
   199→	// Check for witness (witnesses don't have clones, just the witness directory)
   200→	witnessPath := filepath.Join(rigPath, "witness")
   201→	if info, err := os.Stat(witnessPath); err == nil && info.IsDir() {
   202→		rig.HasWitness = true
   203→	}
   204→
   205→	// Check for refinery
   206→	refineryPath := filepath.Join(rigPath, "refinery", "rig")
   207→	if _, err := os.Stat(refineryPath); err == nil {
   208→		rig.HasRefinery = true
   209→	}
   210→
   211→	// Check for mayor clone
   212→	mayorPath := filepath.Join(rigPath, "mayor", "rig")
   213→	if _, err := os.Stat(mayorPath); err == nil {
   214→		rig.HasMayor = true
   215→	}
   216→
   217→	return rig, nil
   218→}

   220→// AddRigOptions configures rig creation.
   221→type AddRigOptions struct {
   222→	Name          string // Rig name (directory name)
   223→	GitURL        string // Repository URL (fetch/pull)
   224→	PushURL       string // Optional push URL (fork for read-only upstreams)
   225→	BeadsPrefix   string // Beads issue prefix (defaults to derived from name)
   226→	LocalRepo     string // Optional local repo for reference clones
   227→	DefaultBranch string // Default branch (defaults to auto-detected from remote)
   228→}

   230→func resolveLocalRepo(path, gitURL string) (string, string) {
   231→	if path == "" {
   232→		return "", ""
   233→	}
   234→
   235→	absPath, err := filepath.Abs(path)
   236→	if err != nil {
   237→		return "", fmt.Sprintf("local repo path invalid: %v", err)
   238→	}
   239→
   240→	absPath, err = filepath.EvalSymlinks(absPath)
   241→	if err != nil {
   242→		return "", fmt.Sprintf("local repo path invalid: %v", err)
   243→	}
   244→
   245→	repoGit := git.NewGit(absPath)
   246→	if !repoGit.IsRepo() {
   247→		return "", fmt.Sprintf("local repo is not a git repository: %s", absPath)
   248→	}
   249→
   250→	origin, err := repoGit.RemoteURL("origin")
   251→	if err != nil {
   252→		return absPath, "local repo has no origin; using it anyway"
   253→	}
   254→	if origin != gitURL {
   255→		return "", fmt.Sprintf("local repo origin %q does not match %q", origin, gitURL)
   256→	}
   257→
   258→	return absPath, ""
   259→}

   261→// AddRig creates a new rig as a container with clones for each agent.
   262→// The rig structure is:
   263→//
   264→//	<name>/                    # Container (NOT a git clone)
   265→//	├── config.json            # Rig configuration
   266→//	├── .beads/                # Rig-level issue tracking
   267→//	├── refinery/rig/          # Canonical main clone
   268→//	├── mayor/rig/             # Mayor's working clone
   269→//	├── witness/               # Witness agent (no clone)
   270→//	├── polecats/              # Worker directories (empty)
   271→//	└── crew/<crew>/           # Default human workspace
   272→func (m *Manager) AddRig(opts AddRigOptions) (*Rig, error) {
   273→	if m.RigExists(opts.Name) {
   274→		return nil, ErrRigExists
   275→	}
   276→
   277→	// Validate rig name: reject characters that break agent ID parsing
   278→	// Agent IDs use format <prefix>-<rig>-<role>[-<name>] with hyphens as delimiters
   279→	if strings.ContainsAny(opts.Name, "-. ") {
   280→		sanitized := strings.NewReplacer("-", "_", ".", "_", " ", "_").Replace(opts.Name)
   281→		sanitized = strings.ToLower(sanitized)
   282→		return nil, fmt.Errorf("rig name %q contains invalid characters; hyphens, dots, and spaces are reserved for agent ID parsing. Try %q instead (underscores are allowed)", opts.Name, sanitized)
   283→	}
   284→
   285→	// Reject reserved names that collide with town-level infrastructure.
   286→	// "hq" is special-cased by EnsureMetadata and dolt routing as the town-level alias.
   287→	for _, reserved := range reservedRigNames {
   288→		if strings.EqualFold(opts.Name, reserved) {
   289→			return nil, fmt.Errorf("rig name %q is reserved for town-level infrastructure", opts.Name)
   290→		}
   291→	}
   292→
   293→	rigPath := filepath.Join(m.townRoot, opts.Name)
   294→
   295→	// Check if directory already exists
   296→	if _, err := os.Stat(rigPath); err == nil {
   297→		return nil, fmt.Errorf("directory already exists: %s\n\nTo adopt an existing directory, use --adopt:\n  gt rig add %s --adopt", rigPath, opts.Name)
   298→	}
   299→
   300→	// Track whether user explicitly provided --prefix (before deriving)
   301→	userProvidedPrefix := opts.BeadsPrefix != ""
   302→	opts.BeadsPrefix = strings.TrimSuffix(opts.BeadsPrefix, "-")
   303→
   304→	// Derive defaults
   305→	if opts.BeadsPrefix == "" {
   306→		opts.BeadsPrefix = deriveBeadsPrefix(opts.Name)
   307→	}
   308→
   309→	localRepo, warn := resolveLocalRepo(opts.LocalRepo, opts.GitURL)
   310→	if warn != "" {
   311→		fmt.Printf("  Warning: %s\n", warn)
   312→	}
   313→
   314→	// Create container directory
   315→	if err := os.MkdirAll(rigPath, 0755); err != nil {
   316→		return nil, fmt.Errorf("creating rig directory: %w", err)
   317→	}
   318→
   319→	// Track cleanup on failure (best-effort cleanup)
   320→	cleanup := func() { _ = os.RemoveAll(rigPath) }
   321→	success := false
   322→	defer func() {
   323→		if !success {
   324→			cleanup()
   325→		}
   326→	}()
   327→
   328→	// Create rig config
   329→	rigConfig := &RigConfig{
   330→		Type:      "rig",
   331→		Version:   CurrentRigConfigVersion,
   332→		Name:      opts.Name,
   333→		GitURL:    opts.GitURL,
   334→		PushURL:   opts.PushURL,
   335→		LocalRepo: localRepo,
   336→		CreatedAt: time.Now(),
   337→		Beads: &BeadsConfig{
   338→			Prefix: opts.BeadsPrefix,
   339→		},
   340→	}
   341→	if err := m.saveRigConfig(rigPath, rigConfig); err != nil {
   342→		return nil, fmt.Errorf("saving rig config: %w", err)
   343→	}
   344→
   345→	// Create shared bare repo as source of truth for refinery and polecats.
   346→	// This allows refinery to see polecat branches without pushing to remote.
   347→	// Mayor remains a separate clone (doesn't need branch visibility).
   348→	fmt.Printf("  Cloning repository (this may take a moment)...\n")
   349→	bareRepoPath := filepath.Join(rigPath, ".repo.git")
   350→	if localRepo != "" {
   351→		if err := m.git.CloneBareWithReference(opts.GitURL, bareRepoPath, localRepo); err != nil {
   352→			fmt.Printf("  Warning: could not use local repo reference: %v\n", err)
   353→			_ = os.RemoveAll(bareRepoPath)
   354→			if err := m.git.CloneBare(opts.GitURL, bareRepoPath); err != nil {
   355→				return nil, wrapCloneError(err, opts.GitURL)
   356→			}
   357→		}
   358→	} else {
   359→		if err := m.git.CloneBare(opts.GitURL, bareRepoPath); err != nil {
   360→			return nil, wrapCloneError(err, opts.GitURL)
   361→		}
   362→	}
   363→	fmt.Printf("   ✓ Created shared bare repo\n")
   364→	bareGit := git.NewGitWithDir(bareRepoPath, "")
   365→
   366→	// Detect empty repos (no commits) early with a clear diagnostic.
   367→	// An empty repo has no refs, so RemoteDefaultBranch/DefaultBranch would
   368→	// return "main" as a fallback, but checkout would fail with an opaque error.
   369→	if empty, err := bareGit.IsEmpty(); err != nil {
   370→		return nil, fmt.Errorf("checking if repository is empty: %w", err)
   371→	} else if empty {
   372→		return nil, fmt.Errorf("repository %s is empty (no commits). Push at least one commit before adding it as a rig", opts.GitURL)
   373→	}
   374→
   375→	// Configure push URL if provided (for read-only upstream repos)
   376→	// This sets origin's push URL to the fork while keeping fetch URL as upstream
   377→	if opts.PushURL != "" {
   378→		if err := bareGit.ConfigurePushURL("origin", opts.PushURL); err != nil {
   379→			return nil, fmt.Errorf("configuring push URL: %w", err)
   380→		}
   381→		fmt.Printf("   ✓ Configured push URL (fork: %s)\n", util.RedactURL(opts.PushURL)) // fmt.Printf matches AddRig's established success output pattern
   382→	}
   383→
   384→	// Determine default branch: use provided value or auto-detect from remote
   385→	var defaultBranch string
   386→	if opts.DefaultBranch != "" {
   387→		defaultBranch = opts.DefaultBranch
   388→	} else {
   389→		// Try to get default branch from remote first, fall back to local detection
   390→		defaultBranch = bareGit.RemoteDefaultBranch()
   391→		if defaultBranch == "" {
   392→			defaultBranch = bareGit.DefaultBranch()
   393→		}
   394→	}
   395→	// When user specified --default-branch, the shallow single-branch clone may not
   396→	// have that branch (it only clones the remote HEAD). Fetch it explicitly.
   397→	if opts.DefaultBranch != "" {
   398→		ref := fmt.Sprintf("origin/%s", defaultBranch)
   399→		if exists, _ := bareGit.RefExists(ref); !exists {
   400→			// Branch not in shallow clone — fetch just that branch
   401→			if err := bareGit.FetchBranchShallow("origin", defaultBranch); err != nil {
   402→				return nil, fmt.Errorf("branch %q does not exist on remote or could not be fetched: %w", defaultBranch, err)
   403→			}
   404→		}
   405→	}
   406→
   407→	rigConfig.DefaultBranch = defaultBranch
   408→	// Re-save config with default branch
   409→	if err := m.saveRigConfig(rigPath, rigConfig); err != nil {
   410→		return nil, fmt.Errorf("updating rig config with default branch: %w", err)
   411→	}
   412→
   413→	// Create mayor as regular clone (separate from bare repo).
   414→	// Mayor doesn't need to see polecat branches - that's refinery's job.
   415→	// This also allows mayor to stay on the default branch without conflicting with refinery.
   416→	// Uses --single-branch --depth 1 --branch to clone only the default branch efficiently.
   417→	fmt.Printf("  Creating mayor clone...\n")
   418→	mayorRigPath := filepath.Join(rigPath, "mayor", "rig")
   419→	if err := os.MkdirAll(filepath.Dir(mayorRigPath), 0755); err != nil {
   420→		return nil, fmt.Errorf("creating mayor dir: %w", err)
   421→	}
   422→	if localRepo != "" {
   423→		if err := m.git.CloneBranchWithReference(opts.GitURL, mayorRigPath, defaultBranch, localRepo); err != nil {
   424→			fmt.Printf("  Warning: could not use local repo reference: %v\n", err)
   425→			_ = os.RemoveAll(mayorRigPath)
   426→			if err := m.git.CloneBranch(opts.GitURL, mayorRigPath, defaultBranch); err != nil {
   427→				return nil, fmt.Errorf("cloning for mayor: %w", err)
   428→			}
   429→		}
   430→	} else {
   431→		if err := m.git.CloneBranch(opts.GitURL, mayorRigPath, defaultBranch); err != nil {
   432→			return nil, fmt.Errorf("cloning for mayor: %w", err)
   433→		}
   434→	}
   435→
   436→	// No explicit checkout needed - --branch already checked out the default branch
   437→	mayorGit := git.NewGitWithDir("", mayorRigPath)
   438→	// Configure push URL on mayor clone (separate clone, doesn't inherit from bare repo)
   439→	if opts.PushURL != "" {
   440→		if err := mayorGit.ConfigurePushURL("origin", opts.PushURL); err != nil {
   441→			return nil, fmt.Errorf("configuring mayor push URL: %w", err)
   442→		}
   443→	}
   444→	fmt.Printf("   ✓ Created mayor clone\n")
   445→
   446→	// Check if source repo has tracked .beads/ directory.
   447→	// If so, we need to initialize the database (it doesn't exist after clone since DB files are gitignored).
   448→	sourceBeadsDir := filepath.Join(mayorRigPath, ".beads")
   449→	if _, err := os.Stat(sourceBeadsDir); err == nil {
   450→		// Remove any redirect file that might have been accidentally tracked.
   451→		// Redirect files are runtime/local config and should not be in git.
   452→		// If not removed, they can cause circular redirect warnings during rig setup.
   453→		sourceRedirectFile := filepath.Join(sourceBeadsDir, "redirect")
   454→		_ = os.Remove(sourceRedirectFile) // Ignore error if doesn't exist
   455→
   456→		// Tracked beads exist - try to detect prefix from existing issues
   457→		sourceBeadsConfig := filepath.Join(sourceBeadsDir, "config.yaml")
   458→		if sourcePrefix := detectBeadsPrefixFromConfig(sourceBeadsConfig); sourcePrefix != "" {
   459→			fmt.Printf("  Detected existing beads prefix '%s' from source repo\n", sourcePrefix)
   460→			// Only error on mismatch if user explicitly provided --prefix
   461→			if userProvidedPrefix && strings.TrimSuffix(opts.BeadsPrefix, "-") != strings.TrimSuffix(sourcePrefix, "-") {
   462→				return nil, fmt.Errorf("prefix mismatch: source repo uses '%s' but --prefix '%s' was provided; use --prefix %s to match existing issues", sourcePrefix, opts.BeadsPrefix, sourcePrefix)
   463→			}
   464→			// Use detected prefix (overrides derived prefix)
   465→			opts.BeadsPrefix = sourcePrefix
   466→			rigConfig.Beads.Prefix = sourcePrefix
   467→			// Re-save rig config with detected prefix
   468→			if err := m.saveRigConfig(rigPath, rigConfig); err != nil {
   469→				return nil, fmt.Errorf("updating rig config with detected prefix: %w", err)
   470→			}
   471→		} else {
   472→			// Detection failed (no issues yet) - use derived/provided prefix
   473→			fmt.Printf("  Using prefix '%s' for tracked beads (no existing issues to detect from)\n", opts.BeadsPrefix)
   474→		}
   475→
   476→		// Initialize bd database if runtime files are missing.
   477→		// DB files are gitignored so they won't exist after clone — bd init creates them.
   478→		// bd init --prefix will create the database and auto-import from issues.jsonl.
   479→		//
   480→		// Note: bdDatabaseExists checks for metadata.json which may be tracked in git.
   481→		// When metadata.json exists but the Dolt server database doesn't (fresh clone
   482→		// to a new workspace), we still need to run bd init to create the server-side
   483→		// database and set issue_prefix. Always ensure issue_prefix is set afterward.
   484→		if !bdDatabaseExists(sourceBeadsDir) {
   485→			initArgs := []string{"init"}
   486→			if opts.BeadsPrefix != "" {
   487→				initArgs = append(initArgs, "--prefix", opts.BeadsPrefix)
   488→			}
   489→			initArgs = append(initArgs, "--server")
   490→			cmd := exec.Command("bd", initArgs...)
   491→			cmd.Dir = mayorRigPath
   492→			if output, err := cmd.CombinedOutput(); err != nil {
   493→				fmt.Printf("  Warning: Could not init bd database: %v (%s)\n", err, strings.TrimSpace(string(output)))
   494→			}
   495→			// Drop orphaned beads_<prefix> database if it differs from rigName (gt-sv1h).
   496→			if orphanDB := "beads_" + opts.BeadsPrefix; orphanDB != opts.Name {
   497→				_ = doltserver.RemoveDatabase(m.townRoot, orphanDB, true)
   498→			}
   499→		}
   500→
   501→		// Always ensure issue_prefix and custom types are configured, even when
   502→		// metadata.json was tracked in git (bdDatabaseExists returned true).
   503→		// The tracked metadata.json tells bd HOW to connect but doesn't guarantee

   765→// InitBeads initializes the beads database at rig level.
   766→// The project's .beads/config.yaml determines sync-branch settings.
   767→// Use `bd doctor --fix` in the project to configure sync-branch if needed.
   768→// TODO(bd-yaml): beads config should migrate to JSON (see beads issue)
   769→//
   770→// rigName is the rig's database name (e.g. "gastown"). When non-empty and
   771→// different from the default "beads_<prefix>" database that bd init creates,
   772→// InitBeads drops the orphan database to prevent accumulation (gt-sv1h).
   773→func (m *Manager) InitBeads(rigPath, prefix, rigName string) error {
   774→	// Validate prefix format to prevent command injection from config files
   775→	if !isValidBeadsPrefix(prefix) {
   776→		return fmt.Errorf("invalid beads prefix %q: must be alphanumeric with optional hyphens, start with letter, max 20 chars", prefix)
   777→	}
   778→
   779→	beadsDir := filepath.Join(rigPath, ".beads")
   780→	mayorRigBeads := filepath.Join(rigPath, "mayor", "rig", ".beads")
   781→
   782→	// Check if source repo has tracked .beads/ (cloned into mayor/rig).
   783→	// If so, create a redirect file instead of a new database.
   784→	if _, err := os.Stat(mayorRigBeads); err == nil {
   785→		// Tracked beads exist - create redirect to mayor/rig/.beads
   786→		if err := os.MkdirAll(beadsDir, 0755); err != nil {
   787→			return err
   788→		}
   789→		redirectPath := filepath.Join(beadsDir, "redirect")
   790→		if err := os.WriteFile(redirectPath, []byte("mayor/rig/.beads\n"), 0644); err != nil {
   791→			return fmt.Errorf("creating redirect file: %w", err)
   792→		}
   793→		return nil
   794→	}
   795→
   796→	// No tracked beads - create local database
   797→	if err := os.MkdirAll(beadsDir, 0755); err != nil {
   798→		return err
   799→	}
   800→
   801→	// Build environment with explicit BEADS_DIR to prevent bd from
   802→	// finding a parent directory's .beads/ database
   803→	env := os.Environ()
   804→	filteredEnv := make([]string, 0, len(env)+1)
   805→	for _, e := range env {
   806→		if !strings.HasPrefix(e, "BEADS_DIR=") {
   807→			filteredEnv = append(filteredEnv, e)
   808→		}
   809→	}
   810→	filteredEnv = append(filteredEnv, "BEADS_DIR="+beadsDir)
   811→
   812→	// Run bd init if available (Dolt is the only backend since bd v0.51.0).
   813→	// --server tells bd to set dolt_mode=server in metadata.json so bd
   814→	// connects to the centralized Dolt sql-server instead of embedded mode.
   815→	initArgs := []string{"init"}
   816→	if prefix != "" {
   817→		initArgs = append(initArgs, "--prefix", prefix)
   818→	}
   819→	initArgs = append(initArgs, "--server")
   820→	cmd := exec.Command("bd", initArgs...)
   821→	cmd.Dir = rigPath
   822→	cmd.Env = filteredEnv
   823→	_, bdInitErr := cmd.CombinedOutput()
   824→	if bdInitErr != nil {
   825→		// bd might not be installed or failed — the shared helper below will
   826→		// create config.yaml with the required defaults as a fallback.
   827→	} else {
   828→		// bd init succeeded - configure the Dolt database
   829→
   830→		// Configure custom types for Gas Town (agent, role, rig, convoy).
   831→		// These were extracted from beads core in v0.46.0 and now require explicit config.
   832→		configCmd := exec.Command("bd", "config", "set", "types.custom", constants.BeadsCustomTypes)
   833→		configCmd.Dir = rigPath
   834→		configCmd.Env = filteredEnv
   835→		// Ignore errors - older beads versions don't need this
   836→		_, _ = configCmd.CombinedOutput()
   837→
   838→		// Explicitly set issue_prefix config (bd init --prefix may not persist it in newer versions).
   839→		// Without this, bd create and gt sling fail with "issue_prefix config is missing".
   840→		prefixSetCmd := exec.Command("bd", "config", "set", "issue_prefix", prefix)
   841→		prefixSetCmd.Dir = rigPath
   842→		prefixSetCmd.Env = filteredEnv
   843→		if prefixOutput, prefixErr := prefixSetCmd.CombinedOutput(); prefixErr != nil {
   844→			return fmt.Errorf("bd config set issue_prefix failed: %s", strings.TrimSpace(string(prefixOutput)))
   845→		}
   846→
   847→		// Default to Dolt-native sync mode for new Gas Town rig beads.
   848→		// Non-fatal: the shared helper below will persist sync.mode to config.yaml.
   849→		syncModeCmd := exec.Command("bd", "config", "set", "sync.mode", "dolt-native")
   850→		syncModeCmd.Dir = rigPath
   851→		syncModeCmd.Env = filteredEnv
   852→		if syncModeOutput, syncModeErr := syncModeCmd.CombinedOutput(); syncModeErr != nil {
   853→			fmt.Printf("   ⚠ Could not set sync.mode via bd: %s\n", strings.TrimSpace(string(syncModeOutput)))
   854→		}
   855→
   856→		// Drop the orphaned beads_<prefix> database created by bd init (gt-sv1h).
   857→		// bd init --prefix creates a database named beads_<prefix> on the Dolt server,
   858→		// but the rig uses <rigName> as its database (set by InitRig + EnsureMetadata).
   859→		// Without cleanup, orphans accumulate with every polecat spawn.
   860→		if rigName != "" {
   861→			orphanDB := "beads_" + prefix
   862→			if orphanDB != rigName {
   863→				_ = doltserver.RemoveDatabase(m.townRoot, orphanDB, true)
   864→			}
   865→		}
   866→	}
   867→
   868→	if err := beads.EnsureConfigYAML(beadsDir, prefix); err != nil {
   869→		return fmt.Errorf("ensuring config.yaml: %w", err)
   870→	}
   871→
   872→	// Ensure database has repository fingerprint (GH #25).
   873→	// This is idempotent - safe on both new and legacy (pre-0.17.5) databases.
   874→	// Without fingerprint, the bd daemon fails to start silently.
   875→	migrateCmd := exec.Command("bd", "migrate", "--update-repo-id")
   876→	migrateCmd.Dir = rigPath
   877→	migrateCmd.Env = filteredEnv
   878→	// Ignore errors - fingerprint is optional for functionality
   879→	_, _ = migrateCmd.CombinedOutput()
   880→
   881→	// Ensure issues.jsonl exists — bd expects this file for git-tracked issue data.
   882→	issuesJSONL := filepath.Join(beadsDir, "issues.jsonl")
   883→	if _, err := os.Stat(issuesJSONL); os.IsNotExist(err) {
   884→		if err := os.WriteFile(issuesJSONL, []byte{}, 0644); err != nil {
   885→			fmt.Printf("   ⚠ Could not create issues.jsonl: %v\n", err)
   886→		}
   887→	}
   888→
   889→	// NOTE: We intentionally do NOT create routes.jsonl in rig beads.
   890→	// bd's routing walks up to find town root (via mayor/town.json) and uses
   891→	// town-level routes.jsonl for prefix-based routing. Rig-level routes.jsonl
   892→	// would prevent this walk-up and break cross-rig routing.
   893→
   894→	return nil
   895→}

   897→// initAgentBeads creates rig-level agent beads for Witness and Refinery.
   898→// These agents use the rig's beads prefix and are stored in rig beads.
   899→//
   900→// Town-level agents (Mayor, Deacon) are created by gt install in town beads.
   901→// Role beads are also created by gt install with hq- prefix.
   902→//
   903→// Rig-level agents (Witness, Refinery) are created here in rig beads with rig prefix.
   904→// Format: <prefix>-<rig>-<role> (e.g., pi-pixelforge-witness)
   905→//
   906→// Agent beads track lifecycle state for ZFC compliance (gt-h3hak, gt-pinkq).
   907→func (m *Manager) initAgentBeads(rigPath, rigName, prefix string) error {
   908→	// Rig-level agents go in rig beads with rig prefix (per docs/architecture.md).
   909→	// Town-level agents (Mayor, Deacon) are created by gt install in town beads.
   910→	// Use ResolveBeadsDir to follow redirect files for tracked beads.
   911→	rigBeadsDir := beads.ResolveBeadsDir(rigPath)
   912→	bd := beads.NewWithBeadsDir(rigPath, rigBeadsDir)
   913→
   914→	// Define rig-level agents to create
   915→	type agentDef struct {
   916→		id       string
   917→		roleType string
   918→		rig      string
   919→		desc     string
   920→	}
   921→
   922→	// Create rig-specific agents using rig prefix in rig beads.
   923→	// Format: <prefix>-<rig>-<role> (e.g., pi-pixelforge-witness)
   924→	agents := []agentDef{
   925→		{
   926→			id:       beads.WitnessBeadIDWithPrefix(prefix, rigName),
   927→			roleType: "witness",
   928→			rig:      rigName,
   929→			desc:     fmt.Sprintf("Witness for %s - monitors polecat health and progress.", rigName),
   930→		},
   931→		{
   932→			id:       beads.RefineryBeadIDWithPrefix(prefix, rigName),
   933→			roleType: "refinery",
   934→			rig:      rigName,
   935→			desc:     fmt.Sprintf("Refinery for %s - processes merge queue.", rigName),
   936→		},
   937→	}
   938→
   939→	// Note: Mayor and Deacon are now created by gt install in town beads.
   940→
   941→	for _, agent := range agents {
   942→		// Check if already exists

   997→// deriveBeadsPrefix generates a beads prefix from a rig name.
   998→// Examples: "gastown" -> "gt", "my-project" -> "mp", "foo" -> "foo"
   999→func deriveBeadsPrefix(name string) string {
  1000→	// Remove common suffixes
  1001→	name = strings.TrimSuffix(name, "-py")
  1002→	name = strings.TrimSuffix(name, "-go")
  1003→
  1004→	// Split on hyphens/underscores
  1005→	parts := strings.FieldsFunc(name, func(r rune) bool {
  1006→		return r == '-' || r == '_'
  1007→	})
  1008→
  1009→	// If single part, try camelCase splitting first (e.g., "myProject" -> "my" + "Project"),
  1010→	// then fall back to compound word detection (e.g., "gastown" -> "gas" + "town").
  1011→	if len(parts) == 1 {
  1012→		if camelParts := splitCamelCase(parts[0]); len(camelParts) >= 2 {
  1013→			parts = camelParts
  1014→		} else {
  1015→			parts = splitCompoundWord(parts[0])
  1016→		}
  1017→	}
  1018→
  1019→	if len(parts) >= 2 {
  1020→		// Take first letter of each part: "gas-town" -> "gt"
  1021→		prefix := ""
  1022→		for _, p := range parts {
  1023→			if len(p) > 0 {
  1024→				prefix += string(p[0])
  1025→			}
  1026→		}
  1027→		return strings.ToLower(prefix)
  1028→	}
  1029→
  1030→	// Single word: use first 2-3 chars
  1031→	if len(name) <= 3 {
  1032→		return strings.ToLower(name)
  1033→	}
  1034→	return strings.ToLower(name[:2])
  1035→}
  1036→
  1037→// splitCompoundWord attempts to split a compound word into its components.
  1038→// Common suffixes like "town", "ville", "port" are detected to split
  1039→// compound names (e.g., "gastown" -> ["gas", "town"]).
  1040→func splitCompoundWord(word string) []string {
  1041→	word = strings.ToLower(word)
  1042→
  1043→	// Common suffixes for compound place names
  1044→	suffixes := []string{"town", "ville", "port", "place", "land", "field", "wood", "ford"}
  1045→
  1046→	for _, suffix := range suffixes {
  1047→		if strings.HasSuffix(word, suffix) && len(word) > len(suffix) {
  1048→			prefix := word[:len(word)-len(suffix)]
  1049→			if len(prefix) > 0 {
  1050→				return []string{prefix, suffix}
  1051→			}
  1052→		}
  1053→	}
  1054→
  1055→	return []string{word}
  1056→}
  1057→
  1058→// splitCamelCase splits a camelCase or PascalCase string into its word parts.
  1059→// Examples: "myProject" -> ["my", "Project"], "gasStation" -> ["gas", "Station"],
  1060→// "HTMLParser" -> ["HTML", "Parser"].
  1061→func splitCamelCase(s string) []string {
  1062→	if s == "" {
  1063→		return nil
  1064→	}
  1065→	var parts []string
  1066→	start := 0
  1067→	runes := []rune(s)
  1068→	for i := 1; i < len(runes); i++ {
  1069→		// Split when transitioning from lower to upper: "myProject" at 'P'
  1070→		if unicode.IsLower(runes[i-1]) && unicode.IsUpper(runes[i]) {
  1071→			parts = append(parts, string(runes[start:i]))
  1072→			start = i
  1073→		}
  1074→		// Split when transitioning from upper run to upper+lower: "HTMLParser" at 'P'
  1075→		if i >= 2 && unicode.IsUpper(runes[i-1]) && unicode.IsUpper(runes[i-2]) && unicode.IsLower(runes[i]) {
  1076→			parts = append(parts, string(runes[start:i-1]))
  1077→			start = i - 1
  1078→		}
  1079→	}
  1080→	parts = append(parts, string(runes[start:]))
  1081→	return parts
  1082→}

  1084→// detectBeadsPrefixFromConfig reads the issue prefix from a beads config.yaml file.
  1085→// Returns empty string if the file doesn't exist or doesn't contain a prefix.
  1086→// Falls back to detecting prefix from existing issues in issues.jsonl.
  1087→//
  1088→// beadsPrefixRegexp validates beads prefix format: alphanumeric, may contain hyphens,
  1089→// must start with letter, max 20 chars. Prevents shell injection via config files.
  1090→var beadsPrefixRegexp = regexp.MustCompile(`^[a-zA-Z][a-zA-Z0-9-]{0,19}$`)
  1091→
  1092→// isValidBeadsPrefix checks if a prefix is safe for use in shell commands.
  1093→// Prefixes must be alphanumeric (with optional hyphens), start with a letter,
  1094→// and be at most 20 characters. This prevents command injection from
  1095→// malicious config files.
  1096→func isValidBeadsPrefix(prefix string) bool {
  1097→	return beadsPrefixRegexp.MatchString(prefix)
  1098→}

  1100→// isStandardBeadHash checks if a string looks like a standard 5-char bead hash.
  1101→// Regular bead IDs use a 5-character base32-encoded hash (e.g., "mawit", "z0ixd").
  1102→// This distinguishes regular issues from agent beads (suffix like "witness")
  1103→// and merge requests (10-char suffix).
  1104→func isStandardBeadHash(s string) bool {
  1105→	if len(s) != 5 {
  1106→		return false
  1107→	}
  1108→	for _, c := range s {
  1109→		if !((c >= 'a' && c <= 'z') || (c >= '0' && c <= '9')) {
  1110→			return false
  1111→		}
  1112→	}
  1113→	return true
  1114→}

  1116→// bdDatabaseExists checks if a beads directory has an initialized database
  1117→// that is actually usable (not just tracked metadata from another workspace).
  1118→//
  1119→// For Dolt server mode, metadata.json may be tracked in git with dolt_database
  1120→// pointing to a database that doesn't exist on this Dolt server. In that case,
  1121→// we need to run bd init to create the server-side database.
  1122→func bdDatabaseExists(beadsDir string) bool {
  1123→	metadataPath := filepath.Join(beadsDir, "metadata.json")
  1124→	data, err := os.ReadFile(metadataPath)
  1125→	if err != nil {
  1126→		return false
  1127→	}
  1128→
  1129→	// Parse metadata to check if the referenced Dolt database actually exists.
  1130→	var meta struct {
  1131→		DoltMode     string `json:"dolt_mode"`
  1132→		DoltDatabase string `json:"dolt_database"`
  1133→	}
  1134→	if err := json.Unmarshal(data, &meta); err != nil {
  1135→		return true // Can't parse — assume it exists (backward compat)
  1136→	}
  1137→
  1138→	// For server mode, verify the database exists in .dolt-data/.
  1139→	// metadata.json may be tracked in git from another workspace where
  1140→	// the Dolt server had this database, but this is a fresh server.
  1141→	if meta.DoltMode == "server" && meta.DoltDatabase != "" {
  1142→		// Walk up from beadsDir to find the town root (.dolt-data lives there).
  1143→		townRoot := beads.FindTownRoot(filepath.Dir(beadsDir))
  1144→		if townRoot == "" {
  1145→			return true // Can't find town root — assume it exists
  1146→		}
  1147→		dbDir := filepath.Join(townRoot, ".dolt-data", meta.DoltDatabase)
  1148→		if _, err := os.Stat(dbDir); os.IsNotExist(err) {
  1149→			return false // Database doesn't exist on this server
  1150→		}
  1151→	}
  1152→
  1153→	return true
  1154→}

  1156→// When adding a rig from a source repo that has .beads/ tracked in git (like a project
  1157→// that already uses beads for issue tracking), we need to use that project's existing
  1158→// prefix instead of generating a new one. Otherwise, the rig would have a mismatched
  1159→// prefix and routing would fail to find the existing issues.
  1160→func detectBeadsPrefixFromConfig(configPath string) string {
  1161→	data, err := os.ReadFile(configPath)
  1162→	if err != nil {
  1163→		return ""
  1164→	}
  1165→
  1166→	// Parse YAML-style config (simple line-by-line parsing)
  1167→	// Looking for "issue-prefix: <value>" or "prefix: <value>"
  1168→	lines := strings.Split(string(data), "\n")
  1169→	for _, line := range lines {
  1170→		line = strings.TrimSpace(line)
  1171→		// Skip comments and empty lines
  1172→		if line == "" || strings.HasPrefix(line, "#") {
  1173→			continue
  1174→		}
  1175→		// Check for issue-prefix or prefix key
  1176→		for _, key := range []string{"issue-prefix:", "prefix:"} {
  1177→			if strings.HasPrefix(line, key) {
  1178→				value := strings.TrimSpace(strings.TrimPrefix(line, key))
  1179→				// Remove quotes if present
  1180→				value = strings.Trim(value, `"'`)
  1181→				if value != "" && isValidBeadsPrefix(value) {
  1182→					return strings.TrimSuffix(value, "-")
  1183→				}
  1184→			}
  1185→		}
  1186→	}
  1187→
  1188→	// Fallback: try to detect prefix from existing issues in issues.jsonl
  1189→	// Parse multiple lines and only consider regular issue IDs (5-char hashes)
  1190→	// to avoid misdetecting agent beads like "gt-demo-witness" as prefix "gt-demo".
  1191→	beadsDir := filepath.Dir(configPath)
  1192→	issuesPath := filepath.Join(beadsDir, "issues.jsonl")
  1193→	if issuesData, err := os.ReadFile(issuesPath); err == nil {
  1194→		issuesLines := strings.Split(string(issuesData), "\n")
  1195→		var detectedPrefix string
  1196→		for _, line := range issuesLines {
  1197→			line = strings.TrimSpace(line)
  1198→			if line == "" {
  1199→				continue
  1200→			}
  1201→			// Look for "id":"<prefix>-<hash>" pattern
  1202→			if idx := strings.Index(line, `"id":"`); idx != -1 {
  1203→				start := idx + 6 // len(`"id":"`)
  1204→				if end := strings.Index(line[start:], `"`); end != -1 {
  1205→					issueID := line[start : start+end]
  1206→					// Only consider IDs with a 5-char alphanumeric hash suffix
  1207→					// (standard bead format). This filters out agent beads
  1208→					// (gt-demo-witness), merge requests (gt-mr-abc1234567),
  1209→					// and other multi-hyphen IDs.
  1210→					if dashIdx := strings.LastIndex(issueID, "-"); dashIdx > 0 {
  1211→						hash := issueID[dashIdx+1:]
  1212→						if !isStandardBeadHash(hash) {
  1213→							continue
  1214→						}
  1215→						prefix := issueID[:dashIdx]
  1216→						if !isValidBeadsPrefix(prefix) {
  1217→							continue
  1218→						}
  1219→						if detectedPrefix == "" {
  1220→							detectedPrefix = prefix
  1221→						} else if detectedPrefix != prefix {
  1222→							// Conflicting prefixes — can't determine reliably
  1223→							return ""
  1224→						}
  1225→					}
  1226→				}
  1227→			}
  1228→		}
  1229→		if detectedPrefix != "" {
  1230→			return detectedPrefix
  1231→		}
  1232→	}
  1233→
  1234→	return ""
  1235→}

  1266→// RegisterRig registers an existing rig directory with the town.
  1267→// Complementary to AddRig: while AddRig creates a new rig from scratch,
  1268→// RegisterRig adopts an existing directory structure.
  1269→func (m *Manager) RegisterRig(opts RegisterRigOptions) (*RegisterRigResult, error) {
  1270→	if m.RigExists(opts.Name) {
  1271→		return nil, ErrRigExists
  1272→	}
  1273→
  1274→	if strings.ContainsAny(opts.Name, "-. ") {
  1275→		sanitized := strings.NewReplacer("-", "_", ".", "_", " ", "_").Replace(opts.Name)
  1276→		sanitized = strings.ToLower(sanitized)
  1277→		return nil, fmt.Errorf("rig name %q contains invalid characters; hyphens, dots, and spaces are reserved for agent ID parsing. Try %q instead (underscores are allowed)", opts.Name, sanitized)
  1278→	}
  1279→
  1280→	for _, reserved := range reservedRigNames {
  1281→		if strings.EqualFold(opts.Name, reserved) {
  1282→			return nil, fmt.Errorf("rig name %q is reserved for town-level infrastructure", opts.Name)
  1283→		}
  1284→	}
  1285→
  1286→	rigPath := filepath.Join(m.townRoot, opts.Name)
  1287→
  1288→	info, err := os.Stat(rigPath)
  1289→	if os.IsNotExist(err) {
  1290→		return nil, fmt.Errorf("directory does not exist: %s", rigPath)
  1291→	}
  1292→	if err != nil {
  1293→		return nil, fmt.Errorf("checking directory: %w", err)
  1294→	}
  1295→	if !info.IsDir() {
  1296→		return nil, fmt.Errorf("not a directory: %s", rigPath)
  1297→	}
  1298→
  1299→	result := &RegisterRigResult{Name: opts.Name}
  1300→
  1301→	// Try to load existing config.json
  1302→	existingConfig, err := LoadRigConfig(rigPath)
  1303→	if err == nil && existingConfig != nil {
  1304→		result.FromConfig = true
  1305→		if opts.GitURL == "" {
  1306→			result.GitURL = existingConfig.GitURL
  1307→		}
  1308→		if opts.BeadsPrefix == "" && existingConfig.Beads != nil {
  1309→			result.BeadsPrefix = existingConfig.Beads.Prefix
  1310→		}
  1311→		result.DefaultBranch = existingConfig.DefaultBranch
  1312→	}
  1313→
  1314→	// If no git URL, try to detect from git remote
  1315→	if result.GitURL == "" && opts.GitURL == "" {
  1316→		detectedURL, detectErr := m.detectGitURL(rigPath)
  1317→		if detectErr != nil && !opts.Force {
  1318→			return nil, fmt.Errorf("could not detect git URL (use --url to specify, or --force to skip): %w", detectErr)
  1319→		}
  1320→		result.GitURL = detectedURL
  1321→	}
  1322→	if opts.GitURL != "" {
  1323→		result.GitURL = opts.GitURL
  1324→	}
  1325→
  1326→	// Derive beads prefix
  1327→	if result.BeadsPrefix == "" && opts.BeadsPrefix == "" {
  1328→		result.BeadsPrefix = deriveBeadsPrefix(opts.Name)
  1329→	}
  1330→	if opts.BeadsPrefix != "" {
  1331→		result.BeadsPrefix = opts.BeadsPrefix
  1332→	}
  1333→
  1334→	// Determine push URL: explicit option > existing config > auto-detect from remotes.
  1335→	// Only explicit option and config.json with non-empty push_url are "authoritative"
  1336→	// (trusted for clearing decisions). Auto-detection runs when no authoritative source
  1337→	// provides a push URL — this covers both fresh adopts and legacy configs that predate
  1338→	// the push_url feature. Auto-detection may fail silently (returns empty on git errors)
  1339→	// and must not trigger stale URL clearing.
  1340→	pushURL := ""
  1341→	pushURLAuthoritative := false // whether the source can be trusted for clearing decisions
  1342→	if opts.PushURL != "" {
  1343→		pushURL = opts.PushURL
  1344→		pushURLAuthoritative = true
  1345→	} else if existingConfig != nil && existingConfig.PushURL != "" {
  1346→		// Config.json has an explicit push URL — use it as authoritative
  1347→		pushURL = existingConfig.PushURL
  1348→		pushURLAuthoritative = true
  1349→	} else {
  1350→		// No authoritative push URL source: either no config.json (fresh adopt) or
  1351→		// legacy config without push_url field. Auto-detect from existing git remotes.
  1352→		pushURL = m.detectPushURL(rigPath)
  1353→		// Not authoritative — only use for positive detection, never for clearing
  1354→	}
  1355→
  1356→	// Apply push URL to existing repos (mirrors AddRig behavior).
  1357→	bareRepoPath := filepath.Join(rigPath, ".repo.git")
  1358→	mayorRigPath := filepath.Join(rigPath, "mayor", "rig")
  1359→	if pushURL != "" {
  1360→		if _, err := os.Stat(bareRepoPath); err == nil {
  1361→			bareGit := git.NewGitWithDir(bareRepoPath, "")
  1362→			if cfgErr := bareGit.ConfigurePushURL("origin", pushURL); cfgErr != nil {
  1363→				return nil, fmt.Errorf("configuring push URL on bare repo: %w", cfgErr)
  1364→			}
  1365→		}
  1366→		if _, err := os.Stat(mayorRigPath); err == nil {
  1367→			mayorGit := git.NewGit(mayorRigPath)
  1368→			if cfgErr := mayorGit.ConfigurePushURL("origin", pushURL); cfgErr != nil {
  1369→				return nil, fmt.Errorf("configuring mayor push URL: %w", cfgErr)
  1370→			}
  1371→		}
  1372→	} else if pushURLAuthoritative {
  1373→		// Clear stale push URLs only when an authoritative source says "no push URL".
  1374→		// Auto-detection returning empty could be a git error — don't clear in that case.
  1375→		// Note: currently unreachable — authoritative sources always set non-empty pushURL.
  1376→		// Retained for future --no-push-url flag support.
  1377→		if _, err := os.Stat(bareRepoPath); err == nil {
  1378→			bareGit := git.NewGitWithDir(bareRepoPath, "")
  1379→			if clrErr := bareGit.ClearPushURL("origin"); clrErr != nil {
  1380→				return nil, fmt.Errorf("clearing stale push URL on bare repo: %w", clrErr)
  1381→			}
  1382→		}
  1383→		if _, err := os.Stat(mayorRigPath); err == nil {
  1384→			mayorGit := git.NewGit(mayorRigPath)
  1385→			if clrErr := mayorGit.ClearPushURL("origin"); clrErr != nil {
  1386→				return nil, fmt.Errorf("clearing stale mayor push URL: %w", clrErr)
  1387→			}
  1388→		}
  1389→	}
  1390→
  1391→	// Sync push URL to config.json so doctor check sees it
  1392→	if existingConfig != nil && existingConfig.PushURL != pushURL {
  1393→		existingConfig.PushURL = pushURL
  1394→		if saveErr := m.saveRigConfig(rigPath, existingConfig); saveErr != nil {
  1395→			// Non-fatal: town.json has the value, but doctor may flag a mismatch
  1396→			fmt.Fprintf(os.Stderr, "Warning: could not update config.json with push URL: %v\n", saveErr)
  1397→		}
  1398→	}
  1399→
  1400→	// Register in town config
  1401→	m.config.Rigs[opts.Name] = config.RigEntry{
  1402→		GitURL:  result.GitURL,
  1403→		PushURL: pushURL,
  1404→		AddedAt: time.Now(),
  1405→		BeadsConfig: &config.BeadsConfig{
  1406→			Prefix: result.BeadsPrefix,
  1407→		},
  1408→	}
  1409→
  1410→	return result, nil
  1411→}

  1413→// detectPushURL attempts to detect a custom push URL from an existing repository.
  1414→// Returns empty string if push URL matches fetch URL (no custom push URL configured).
  1415→func (m *Manager) detectPushURL(rigPath string) string {
  1416→	// Check bare repo first (polecat-preferred source of truth), then clones.
  1417→	// .repo.git is a bare repo and requires NewGitWithDir; the rest are regular clones.
  1418→	bareRepoPath := filepath.Join(rigPath, ".repo.git")
  1419→	if pushURL := detectPushURLFrom(git.NewGitWithDir(bareRepoPath, "")); pushURL != "" {
  1420→		return pushURL
  1421→	}
  1422→
  1423→	clonePaths := []string{
  1424→		rigPath,
  1425→		filepath.Join(rigPath, "mayor", "rig"),
  1426→		filepath.Join(rigPath, "refinery", "rig"),
  1427→	}
  1428→	for _, p := range clonePaths {
  1429→		if pushURL := detectPushURLFrom(git.NewGit(p)); pushURL != "" {
  1430→			return pushURL
  1431→		}
  1432→	}
  1433→	return ""
  1434→}

  1436→// detectPushURLFrom checks a single git repo for a custom push URL.
  1437→func detectPushURLFrom(g *git.Git) string {
  1438→	fetchURL, fetchErr := g.RemoteURL("origin")
  1439→	if fetchErr != nil {
  1440→		return ""
  1441→	}
  1442→	pushURL, pushErr := g.GetPushURL("origin")
  1443→	if pushErr != nil || pushURL == "" {
  1444→		return ""
  1445→	}
  1446→	if strings.TrimSpace(pushURL) != strings.TrimSpace(fetchURL) {
  1447→		return strings.TrimSpace(pushURL)
  1448→	}
  1449→	return ""
  1450→}

  1452→// detectGitURL attempts to detect the git remote URL from an existing repository.
  1453→// detectGitURL finds the origin remote URL from available clones.
  1454→// Note: .repo.git is intentionally not checked here — it's a bare repo shared by worktrees
  1455→// and requires NewGitWithDir (not NewGit). detectPushURL checks .repo.git because push URL
  1456→// is primarily configured there. For git URL, the clone-based paths are authoritative.
  1457→func (m *Manager) detectGitURL(rigPath string) (string, error) {
  1458→	possiblePaths := []string{
  1459→		rigPath,
  1460→		filepath.Join(rigPath, "mayor", "rig"),
  1461→		filepath.Join(rigPath, "refinery", "rig"),
  1462→	}
  1463→	for _, p := range possiblePaths {
  1464→		g := git.NewGit(p)
  1465→		url, err := g.RemoteURL("origin")
  1466→		if err == nil && url != "" {
  1467→			return strings.TrimSpace(url), nil
  1468→		}
  1469→	}
  1470→	return "", fmt.Errorf("no git repository with origin remote found in %s", rigPath)
  1471→}

  1481→// seedPatrolMolecules creates patrol molecule prototypes in the rig's beads database.

  1483→func (m *Manager) seedPatrolMolecules(rigPath string) error {

  1495→// seedPatrolMoleculesManually creates patrol molecules using bd create commands.
  1496→func (m *Manager) seedPatrolMoleculesManually(rigPath string) error {
  1497→	// Patrol molecule definitions for seeding
  1498→	patrolMols := []struct {
  1499→		title string
  1500→		desc  string
  1501→	}{
  1502→		{
  1503→			title: "Deacon Patrol",
  1504→			desc:  "Mayor's daemon patrol loop for handling callbacks, health checks, and cleanup.",
  1505→		},
  1506→		{
  1507→			title: "Witness Patrol",
  1508→			desc:  "Per-rig worker monitor patrol loop with progressive nudging.",
  1509→		},
  1510→		{
  1511→			title: "Refinery Patrol",
  1512→			desc:  "Merge queue processor patrol loop with verification gates.",
  1513→		},
  1514→	}
  1515→
  1516→	for _, mol := range patrolMols {
  1517→		// Check if already exists by title
  1518→		checkCmd := exec.Command("bd", "list", "--type=molecule", "--format=json")
  1519→		checkCmd.Dir = rigPath
  1520→		output, _ := checkCmd.Output()
  1521→		if strings.Contains(string(output), mol.title) {
  1522→			continue // Already exists
  1523→		}
  1524→
  1525→		// Create the molecule
  1526→		cmd := exec.Command("bd", "create", //nolint:gosec // G204: bd is a trusted internal tool
  1527→			"--type=molecule",
  1528→			"--title="+mol.title,
  1529→			"--description="+mol.desc,
  1530→			"--priority=2",
  1531→		)
  1532→		cmd.Dir = rigPath
  1533→		if err := cmd.Run(); err != nil {
  1534→			// Non-fatal, continue with others
  1535→			continue
  1536→		}
  1537→	}
  1538→	return nil
  1539→}
