   380‚Üífunc confirmUnsafeProceed(force bool) bool {
   381‚Üí	// If --force and interactive TTY, prompt.
   382‚Üí	if force && isStdinTerminal() {
   383‚Üí		fmt.Println()
   384‚Üí		return promptYesNoUnsafeProceed("Proceed anyway?")
   385‚Üí	}
   386‚Üí
   387‚Üí	// Otherwise block with hint.
   388‚Üí	if force {
   389‚Üí		fmt.Printf("\n%s requires an interactive terminal. Use %s to skip all checks (DANGER: will lose work!)\n",
   390‚Üí			style.Bold.Render("--force"), style.Bold.Render("--nuclear"))
   391‚Üí	} else {
   392‚Üí		fmt.Printf("\nUse %s to proceed with confirmation, or %s to skip all checks (DANGER: will lose work!)\n",
   393‚Üí			style.Bold.Render("--force"), style.Bold.Render("--nuclear"))
   394‚Üí	}
   395‚Üí	return false
   396‚Üí}

   398‚Üí// checkUncommittedWork checks polecats in a rig for uncommitted work.
   399‚Üí// operation is the verb shown in the warning (e.g. "stop", "shutdown", "restart").
   400‚Üí// Returns true if the caller should proceed, false if it should abort.
   401‚Üí// When force is true and stdin is a TTY, prompts the user to confirm.
   402‚Üí// When force is true but stdin is NOT a TTY, blocks (same as no --force).
   403‚Üí// All user-facing messages are printed internally.
   404‚Üífunc checkUncommittedWork(r *rig.Rig, rigName, operation string, force bool) (proceed bool) {
   405‚Üí	polecats, err := listPolecatsForWorkCheck(r)
   406‚Üí	if err != nil {
   407‚Üí		fmt.Printf("%s Could not check polecats for uncommitted work: %v\n",
   408‚Üí			style.Warning.Render("‚öÝ"), err)
   409‚Üí		return confirmUnsafeProceed(force)
   410‚Üí	}
   411‚Üí	if len(polecats) == 0 {
   412‚Üí		return true
   413‚Üí	}
   414‚Üí
   415‚Üí	var problemPolecats []struct {
   416‚Üí		name   string
   417‚Üí		status *git.UncommittedWorkStatus
   418‚Üí	}
   419‚Üí	var checkErrors []struct {
   420‚Üí		name string

  1131‚Üí// runResetStale resets in_progress issues whose assigned agent no longer has a session.
  1132‚Üífunc runResetStale(bd *beads.Beads, dryRun bool) error {
  1133‚Üí	t := tmux.NewTmux()
  1134‚Üí
  1135‚Üí	// Get all in_progress issues
  1136‚Üí	issues, err := bd.List(beads.ListOptions{
  1137‚Üí		Status:   "in_progress",
  1138‚Üí		Priority: -1, // All priorities

  1148‚Üí
  1149‚Üí	var resetCount, skippedCount int
  1150‚Üí	var resetIssues []string
  1151‚Üí
  1152‚Üí	for _, issue := range issues {
  1153‚Üí		if issue.Assignee == "" {

  1156‚Üí
  1157‚Üí		// Parse assignee: rig/name or rig/crew/name
  1158‚Üí		sessionName, isPersistent := assigneeToSessionName(issue.Assignee)
  1159‚Üí		if sessionName == "" {
  1160‚Üí			continue // Couldn't parse assignee
  1161‚Üí		}
  1162‚Üí
  1163‚Üí		// Check if session exists
  1164‚Üí		hasSession, err := t.HasSession(sessionName)
  1165‚Üí		if err != nil {
  1166‚Üí			// tmux error, skip this one
  1167‚Üí			continue
  1168‚Üí		}
  1169‚Üí
  1170‚Üí		if hasSession {
  1171‚Üí			continue // Session exists, not stale
  1172‚Üí		}
  1173‚Üí
  1174‚Üí		// For crew (persistent identities), only reset if explicitly checking sessions
  1175‚Üí		if isPersistent {
  1176‚Üí			skippedCount++
  1177‚Üí			if dryRun {
  1178‚Üí				fmt.Printf("  %s: %s %s\n",
  1179‚Üí					style.Dim.Render(issue.ID),
  1180‚Üí					issue.Assignee,
  1181‚Üí					style.Dim.Render("(persistent, skipped)"))
  1182‚Üí			}
  1183‚Üí			continue
  1184‚Üí		}
  1185‚Üí
  1186‚Üí		// Session doesn't exist - this is stale
  1187‚Üí		if dryRun {
  1188‚Üí			fmt.Printf("  %s: %s (no session) ‚Üí open\n",
  1189‚Üí				style.Bold.Render(issue.ID),
  1190‚Üí				issue.Assignee)
  1191‚Üí		} else {
  1192‚Üí			// Reset status to open and clear assignee
  1193‚Üí			openStatus := "open"
  1194‚Üí			emptyAssignee := ""
  1195‚Üí			if err := bd.Update(issue.ID, beads.UpdateOptions{
  1196‚Üí				Status:   &openStatus,
  1197‚Üí				Assignee: &emptyAssignee,
  1198‚Üí			}); err != nil {
  1199‚Üí				fmt.Printf("  %s Failed to reset %s: %v\n",
  1200‚Üí					style.Warning.Render("‚öÝ"),
  1201‚Üí					issue.ID, err)
  1202‚Üí				continue
  1203‚Üí			}
  1204‚Üí		}
  1205‚Üí		resetCount++
  1206‚Üí		resetIssues = append(resetIssues, issue.ID)
  1207‚Üí	}
  1208‚Üí
  1209‚Üí	if dryRun {
  1210‚Üí		if resetCount > 0 || skippedCount > 0 {
  1211‚Üí			fmt.Printf("\n%s Would reset %d issues, skip %d persistent\n",
  1212‚Üí				style.Dim.Render("(dry-run)"),
  1213‚Üí				resetCount, skippedCount)
  1214‚Üí		} else {
  1215‚Üí			fmt.Printf("%s No stale issues found\n", style.Success.Render("‚úì"))
  1216‚Üí		}
  1217‚Üí	} else {
  1218‚Üí		if resetCount > 0 {
  1219‚Üí			fmt.Printf("%s Reset %d stale issues: %v\n",
  1220‚Üí				style.Success.Render("‚úì"),
  1221‚Üí				resetCount, resetIssues)
  1222‚Üí		} else {
  1223‚Üí			fmt.Printf("%s No stale issues to reset\n", style.Success.Render("‚úì"))
  1224‚Üí		}

  1949‚Üí// getRigOperationalState returns the operational state and source for a rig.

  1951‚Üí// Returns state ("OPERATIONAL", "PARKED", or "DOCKED") and source ("local", "global - synced", or "default").

  1976‚Üí			for _, label := range issue.Labels {

  1979‚Üí					switch strings.ToLower(statusValue) {
