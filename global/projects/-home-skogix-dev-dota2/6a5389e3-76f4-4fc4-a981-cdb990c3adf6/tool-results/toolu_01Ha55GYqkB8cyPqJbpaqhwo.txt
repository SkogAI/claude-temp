   425→function buildHeroTable(data) {
   426→  var tbody = document.getElementById('hero-tbody');
   427→  var filtered = data;
   428→  if (currentFilter === 'str') filtered = data.filter(function(h){return h.attr === 'str';});
   429→  else if (currentFilter === 'agi') filtered = data.filter(function(h){return h.attr === 'agi';});
   430→  else if (currentFilter === 'int') filtered = data.filter(function(h){return h.attr === 'int';});
   431→  else if (currentFilter === '10+') filtered = data.filter(function(h){return h.games >= 10;});
   432→  else if (currentFilter === 'wr55+') filtered = data.filter(function(h){return h.games >= 10 && h.wr >= 55;});
   433→
   434→  var html = '';
   435→  filtered.forEach(function(h) {
   436→    var wrCls = h.wr >= 55 ? 'wr-good' : h.wr < 45 ? 'wr-bad' : 'wr-neutral';
   437→    var attrColor = h.attr === 'str' ? 'var(--str)' : h.attr === 'agi' ? 'var(--agi)' : h.attr === 'int' ? 'var(--int)' : 'var(--all)';
   438→    html += '<tr>';
   439→    html += '<td><div class="hero-cell"><img src="' + h.img + '" alt="' + h.name + '">' + h.name + '</div></td>';
   440→    html += '<td style="color:' + attrColor + '">' + h.attr.toUpperCase() + '</td>';
   441→    html += '<td>' + h.games + '</td>';
   442→    html += '<td>' + h.wins + '</td>';
   443→    html += '<td>' + h.losses + '</td>';
   444→    html += '<td class="' + wrCls + '">' + h.wr + '%</td>';
   445→    html += '<td style="color:var(--text-dim)">' + h.lastPlayed + '</td>';
   446→    html += '</tr>';
   447→  });
   448→  tbody.innerHTML = html;
   449→}

   458→function sortTable(col) {
   459→  if (currentSort.col === col) currentSort.asc = !currentSort.asc;
   460→  else { currentSort.col = col; currentSort.asc = col <= 1; }
   461→
   462→  var keys = ['name', 'attr', 'games', 'wins', 'losses', 'wr', 'lastPlayed'];
   463→  var key = keys[col];
   464→  D.heroStats.sort(function(a, b) {
   465→    var va = a[key], vb = b[key];
   466→    if (typeof va === 'string') { va = va.toLowerCase(); vb = vb.toLowerCase(); }
   467→    if (va < vb) return currentSort.asc ? -1 : 1;
   468→    if (va > vb) return currentSort.asc ? 1 : -1;
   469→    return 0;
   470→  });
   471→  buildHeroTable(D.heroStats);
   472→}
